vcard

Created 16 Dec 2021 by SP

Invisible XML Grammar

card: -"BEGIN:", name, eoln, property+, -"END:", endname, eoln.
property: name, parameters, -":", attribute++-";", -eoln.
parameters: (-";", parameter)*.
parameter: name, -"=", pvalue.
@pvalue: ~[";:"; #a; #d]+.
attribute: value.
@value: achar*.
-achar: ~["#;"; #a; #d].
@name: ["a"-"z"; "A"-"Z"]+.
@endname: name.
-eoln: -#a | -#d, -#a.
