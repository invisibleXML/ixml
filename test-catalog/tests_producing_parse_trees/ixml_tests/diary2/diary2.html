Invisible XML Grammar

diary: entry+.
entry: date, para.
date: day, s, month, s,  year, nl.
day: digit, digit?.
-digit:["0"-"9"].
month: "January"; "February"; "March"; "April"; "May"; "June";
       "July"; "August"; "September"; "October"; "November"; "December".
year: digit, digit, digit, digit.

para: char*, blank.
-blank: nl, nl.
-char: letter; punctuation; s.
-letter: [L].
-punctuation: [".;:,'?!"].
-s: " ".
-nl: -#a | -#d, -#a .
